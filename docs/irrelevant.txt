package main

import (
	"fmt"
	"net/http"
)

type Invoice struct {
	InvoiceID string
	Amount float64
	VATRate float64
	Customer string
}

func main() {

	const port = ":8080"

	http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
		fmt.Fprintf(w, "Backend is LIVE - Ready for E-VAT 2026")
	})

	invoice := Invoice{
		InvoiceID: "INV-001",
		Amount: 1000,
		VATRate: 0.15,
		Customer: "ABC Trading Ltd",
	}

	fmt.Println("Sample Invoice Loaded:")
	fmt.Println(invoice)

	fmt.Printf("Server starting on http://localhost%s...\n", port)

	err := http.ListenAndServe(port, nil)
	if err != nil {
		fmt.Printf("Error starting server: %\n", err)
	}
}
